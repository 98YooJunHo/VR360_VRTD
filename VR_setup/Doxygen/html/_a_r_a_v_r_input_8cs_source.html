310 public static void DrawCrosshair(Transform crosshair, bool isHand = true, Controller hand = Controller.RTouch)
311 {
312 Ray ray;
313 // 컨트롤러의 위치와 방향을 이용해 레이 제작
314 if (isHand)
315 {
316#if BUILD_PLATFORM_PC
317 ray = Camera.main.ScreenPointToRay(Input.mousePosition);
318#elif TARGET_DEVICE_OCULUS
319 if(hand == Controller.RTouch)
320 {
321 ray = new Ray(RHandPosition, RHandDirection);
322 } // 오른쪽 컨트롤러에서 레이를 쏘는 경우
323 else
324 {
325 ray = new Ray(LHandPosition, LHandDirection);
326 } // 왼쪽 컨트롤러에서 레이를 쏘는 경우
327#endif
328 }
329 else
330 {
331 // 카메라 기준으로 화면의 정중앙으로 레이를 제작
332 ray = new Ray(Camera.main.transform.position, Camera.main.transform.forward);
333 }
334
335 // 눈에 안보이는 plane을 만든다.
336 Plane plane = new Plane(Vector3.up, 0);
337 float distance = 0;
338 // plane을 이용해 ray를 쏜다
339 if (plane.Raycast(ray, out distance))
340 {
341 // 레이의 GetPoint함수를 이용해 충돌 지점의 위치를 가져온다
342 crosshair.position = ray.GetPoint(distance);
343 crosshair.forward = -Camera.main.transform.forward;
344 // 크로스헤어의 크기를 최소 기본 크기에서 거리에 따라 더 커지도록 한다
345 crosshair.localScale = originScale * Mathf.Max(1, distance);
346 } // ray 충돌
347 else
348 {
349 crosshair.position = ray.origin + ray.direction * 100;
350 crosshair.forward = -Camera.main.transform.forward;
351 distance = (crosshair.position - ray.origin).magnitude;
352 crosshair.localScale = originScale * Mathf.Max(1, distance);
353 } // ray 미충돌
354 }

379 static IEnumerator VibrationCoroutine(float duration, float frequency, float amplitude, Controller hand)
380 {
381 float currentTime = 0;
382 while (currentTime < duration)
383 {
384 currentTime += Time.deltaTime;
385 OVRInput.SetControllerVibration(frequency, amplitude, (OVRInput.Controller)hand);
386 yield return null;
387 } // loop: duration 시간 동안 반복하는 루프
388 OVRInput.SetControllerVibration(0, 0, (OVRInput.Controller)hand);
389 }
